# mn-lottery-grid 九宫格抽奖组件

组件名：mn-lottery-grid 代码块： lottery-grid

# 安装方式

本组件符合easycom (opens new window)规范，HBuilderX 2.5.5起，只需将本组件导入项目，在页面template中即可直接使用，无需在页面中import和注册components。

如需通过npm方式使用mn-mall-ui组件，另见文档：https://ext.dcloud.net.cn/plugin?id=55 (opens new window)

# 基本用法

在 template 中使用组件

<mn-lottery-grid
  ref="lotteryGridRef"
  :raffleNum="raffleNum"
  :award="award"
  :awards="awards"
  @raffle="handleRaffle"
  @over="handleOver"
/>

# API

# mn-lottery-grid: Props 属性

属性名 类型 默认值 说明
raffleNum Number '' 抽奖次数
award Object '#fff' 获奖结果
awards Array 0 奖品列表
ref String 'lotteryGridRef' 引用属性

# mn-lottery-grid: Events

事件名 说明 返回值
raffle 抽奖开始异步 无
over 抽奖结束事件 当前中奖索引 current
handleStart 抽奖开始同步 无

# 组件示例

<mn-lottery-grid
  ref="lotteryGridRef"
  :raffleNum="raffleNum"
  :award="award"
  :awards="awards"
  @raffle="handleRaffle"
  @over="handleOver"
/>

export default {
  data() {
    return {
      raffleNum: 3,
      award: {},
      awards: []
    };
  },
  methods: {
    handleRaffle() {
      if (!this.raffleNum) {
        uni.showToast({
          title: '抽奖次数已经用完～',
          duration: 3000
        });
        return;
      }
      //刷新抽奖结果也是异步
      this.raffleNum -= 1;
      //获取奖品结果是异步
      this.award = {
        prize_type: 1
      };
      this.$refs.lotteryGridRef.handleStart();
    },
    handleOver(current) {
      this.prizeModalStr = this.awards[current].title.split(' ').join('');
      uni.showToast({
        title: this.prizeModalStr,
        duration: 3000
      });
    }
  },
  created() {
    const tempArr = [
      {
        prize_type: 1,
        title: '北京环球影城 门票一张',
        image: ''
      },
      {
        prize_type: 2,
        title: '5元 无门槛',
        image: ''
      },
      {
        prize_type: 3,
        title: '满75减 10元',
        image: ''
      },
      {
        prize_type: 4,
        title: '满110减 15元',
        image: ''
      },
      {
        prize_type: 5,
        title: '满165减 30元',
        image: ''
      },
      {
        prize_type: 6,
        title: '满199减 50元',
        image: ''
      }
    ];
    const star = {
      image: 'https://mn-duoshanghu.oss-cn-beijing.aliyuncs.com/xndj/bfcd6202110211950595766.png'
    };
    tempArr.splice(3, 0, star);
    tempArr.push(star);
    //填充数组
    tempArr.forEach((element, index) => {
      if (element.title) {
        element.title1 = element.title.split(' ')[0];
        element.title2 = element.title.split(' ')[1];
      }
      this.$set(this.awards, index, element);
    });
  }
};
